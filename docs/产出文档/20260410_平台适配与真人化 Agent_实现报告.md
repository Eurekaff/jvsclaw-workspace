##  使用示例

### 1. 平台适配

```python
from agents.platform_adapter.agent import PlatformAdapterAgent

agent = PlatformAdapterAgent("/path/to/workspace")

result = agent.execute({
    "core_content": {
        "topic": "AI 工具推荐",
        "content": "核心内容...",
        "selling_points": ["提高效率", "节省时间"],
        "target_audience": "打工人"
    },
    "platforms": ["xiaohongshu", "douyin", "weibo"]
})

# 获取各平台版本
xiaohongshu_version = result["platform_versions"]["xiaohongshu"]
douyin_script = result["platform_versions"]["douyin"]
```

### 2. 真人化检查

```python
from agents.authenticity_checker.agent import AuthenticityCheckerAgent

agent = AuthenticityCheckerAgent("/path/to/workspace")

result = agent.execute({
    "content": "要检查的内容",
    "platform": "xiaohongshu"
})

report = result["authenticity_report.json"]

# 检查是否通过
if report["passed"]:
    print("✅ 可以发布")
else:
    print("❌ 需要修改")
    print("建议:", report["suggestions"])
```

### 3. 完整工作流

```python
# 1. 生成核心内容
core_content = generate_content(topic)

# 2. 平台适配
adapter = PlatformAdapterAgent()
versions = adapter.execute({
    "core_content": core_content,
    "platforms": ["xiaohongshu", "weibo"]
})

# 3. 真人化检查
checker = AuthenticityCheckerAgent()
for platform, version in versions.items():
    result = checker.execute({
        "content": version["content"]["text"],
        "platform": platform
    })
    
    if result["report"]["passed"]:
        publish(platform, version)
    else:
        revise(version, result["report"]["suggestions"])
```
