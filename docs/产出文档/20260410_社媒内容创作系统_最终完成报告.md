## 🚀 使用示例

### 完整工作流

```python
from agents.trend_hunter.agent import TrendHunterAgent
from agents.content_creator.agent import ContentCreatorAgent
from agents.image_generator.agent import ImageGeneratorAgent
from agents.multi_platform_publisher.agent import MultiPlatformPublisherAgent

# 1. 发现热点
trend_hunter = TrendHunterAgent()
trends = trend_hunter.execute({
    "platforms": ["weibo", "zhihu"],
    "limit": 10
})

# 2. 选择热门选题
hot_topic = trends["recommendations"][0]

# 3. 生成内容
content_creator = ContentCreatorAgent()
content = content_creator.execute({
    "topic": hot_topic["topic"],
    "platforms": ["xiaohongshu", "weibo"],
    "selling_points": hot_topic["keywords"]
})

# 4. 生成配图
image_gen = ImageGeneratorAgent()
images = image_gen.execute({
    "content": content["core_content"],
    "image_types": ["cover", "thumbnail"]
})

# 5. 发布
publisher = MultiPlatformPublisherAgent()
result = publisher.execute({
    "content": content["platform_versions"]["xiaohongshu"],
    "platforms": ["xiaohongshu", "weibo"],
    "images": images["images"],
    "auto_approve": True
})
```
